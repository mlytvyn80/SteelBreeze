00071     case SIGABRT: sigaction(signum, &saveSIGABRT,NULL); break;
00072     case SIGBUS:  sigaction(signum, &saveSIGBUS, NULL); break;

00077     case SIGTERM: sigaction(signum, &saveSIGTERM,NULL); break;
