00566 void SBFilteringGauss::makeAnalysisWith3Sigma()
00567 {
00568   double        f = 1.0;
00569   if (Filter->isOK())
00570     {
00571       for (unsigned int j=0; j<NumOfDataCols; j++)
00572         {
00573           switch (SMode)
00574             {
00575             case SIG_MODE_0: f *= Filter->sigma(  j);   break;
00576             case SIG_MODE_1: f *= Filter->sigma(  j);   break;
00577             case SIG_MODE_2: f *= Filter->sigma(2*j);   break;
00578             };
00579         };
00580       if (f>=0.0)
00581         {
00582           Filter->addAttr(SBFilterGauss::IsOmit3Sigma);
00583           redrawDataPlot();
00584           Filter->delAttr(SBFilterGauss::IsOmit3Sigma);
00585         }
00586       else
00587         Log->write(SBLog::INF, SBLog::DATA, ClassName() + 
00588                    ": cannot calculate interpolation with 3Sigma limits; sigmas are zeros" );
00589     };
