00149 bool SBSource::isValidICRFName(QString sICRF)
00150 {
00151   bool          is = FALSE;
00152   int           v;
00153 
00154   //0123456789012345
00155   //JHHMMSS.s-DDMMSS
00156   if (sICRF.length()!=16) return FALSE;
00157   if (sICRF.mid(0,1)!="J") return FALSE;
00158 
00159   v = sICRF.mid(1, 2).toInt(&is);
00160   if (!is || (0>v) || (v>=24)) return FALSE;
00161   v = sICRF.mid(3, 2).toInt(&is);
00162   if (!is || (0>v) || (v>=60)) return FALSE;
00163   v = sICRF.mid(5, 2).toInt(&is);
00164   if (!is || (0>v) || (v>=60)) return FALSE;
00165   if (sICRF.mid(7,1)!=".") return FALSE;
00166   v = sICRF.mid(8, 1).toInt(&is);
00167   if (!is) return FALSE;
00168   if ((sICRF.mid(9,1)!="+")&&(sICRF.mid(9,1)!="-")) return FALSE;
00169   
00170   v = sICRF.mid(10, 2).toInt(&is);
00171   if (!is || (0>v) || (v>=90)) return FALSE;
00172   v = sICRF.mid(12, 2).toInt(&is);
00173   if (!is || (0>v) || (v>=60)) return FALSE;
00174   v = sICRF.mid(14, 2).toInt(&is);
00175   if (!is || (0>v) || (v>=60)) return FALSE;
00176 
00177   return TRUE;
