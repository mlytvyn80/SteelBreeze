00339 void SBEphem::interp(double T, int Offset, int NCo, int NSets, SBCelestBody& B)
00340 {
00341   // T is (T-T_0)/DT
00342   int           i,j,l;
00343   double        bma, bma2, tc, twot, f;
00344   double        pc[18], vc[18], ac[18];
00345 
00346   pc[0] = 1.0;
00347   vc[1] = 1.0;
00348   ac[2] = 4.0;
00349 
00350   bma   = 2.0*NSets/dt();
00351   bma2  = bma*bma;
00352 
00353   l     = (int)floor(T*NSets - floor(T)); //index of coeff's cluster
00354   tc    = 2.0* (fmod(T*NSets, 1.0) + floor(T)) - 1.0;
00355   twot  = 2.0*tc;
00356 
00357   pc[1] = tc;
00358   vc[2] = 2.0*twot;
00359   ac[3] = 24.0*pc[1];
00360 
00361   for (i=2; i<NCo; i++) pc[i] = twot*pc[i-1] - pc[i-2];
00362   for (i=3; i<NCo; i++) vc[i] = twot*vc[i-1] + 2.0*pc[i-1] - vc[i-2];
00363   for (i=4; i<NCo; i++) ac[i] = twot*ac[i-1] + 4.0*vc[i-1] - ac[i-2];
00364 
00365   for (i=0; i<3; i++)
00366     {
00367       f=0.0;
00368       for (j=0; j<NCo; j++)f += pc[j]**(buff + Offset-1 + j + i*NCo + l*NCo*3);
00369       B.R[(DIRECTION)i]=f;
00370       f=0.0;
00371       for (j=1; j<NCo; j++)f += vc[j]**(buff + Offset-1 + j + i*NCo + l*NCo*3);
00372       B.V[(DIRECTION)i]=f;
00373       f=0.0;
00374       for (j=2; j<NCo; j++)f += ac[j]**(buff + Offset-1 + j + i*NCo + l*NCo*3);
00375       B.A[(DIRECTION)i]=f;
00376     };
00377   B.R*=1000.0;
00378   B.V*=bma*1000.0/DAY2SEC;
00379   B.A*=bma2*1000.0/DAY2SEC/DAY2SEC;
