00131     Assuming that time t1 is Atomic (TAI) time of reception of the VLBI signal at receiver 1,

00133

00138   X2 = Ephem->earth().r() + x2;

00140   /*

00146   double Uj = 0.0;
00147   double U_ = 0.0;
00148 
00149   GravDelay = 0.0;
00150   for (int j=0; j<11; j++)

00152       {

00154         Ephem->calcI(j, t1j);

00160           "
00161           3. Use equation 1 to estimate the differential gravitational
00162           delay of those bodies."
00163         */

00166         U_+=Uj;

00169           {
00170             if (acos(K*(-R1j)/R1j.module())<Cfg->gravHOThreshold())
00171               {
00172                 GravDelay += a*a/cLight()*(b*(R1j.unit() + K))/(R1j.module() + K*R1j)/(R1j.module() + K*R1j);
00173               };
00174             if (acos(K*(-R2j)/R2j.module())<Cfg->gravHOThreshold())
00175               {
00176                 GravDelay += a*a/cLight()*(b*(R2j.unit() + K))/(R2j.module() + K*R2j)/(R2j.module() + K*R2j);
00177               };
00178           };

00180         if (Log->isEligible(SBLog::DBG, SBLog::GRAVDELAY))

00183                        ": Celestial Body: " + Ephem->body(j).name());

00185                        ": Arg(TT)       : " + t1.toString());

00187                        ": t1j(TT)       : " + t1j.toString());

00189                            ": R<1-j>(CRS,km): ", 15, 3, 1.0e-3);

00191                            ": R<2-j>(CRS,km): ", 15, 3, 1.0e-3);

00193                        ": Grav. delay (ps)  : " + QString().sprintf("%16.8f", 1.0e12*GravDelJ));
00194             Log->write(SBLog::DBG, SBLog::GRAVDELAY, ClassName() + 
00195                        ": Potential Uj      : " + QString().sprintf("%30.14f", Uj));
00196           };

00200     "
00201     4. Use equation 2 to find the differential delay due to the Earth.
00202     5. Sum to find the differential gravitational delay."

00226     "

00281

00284

00288                    Frame, St_2, So, Frame->crs2trs()*k2, St_2Info->zw(), IsArtMeteo)/cLight();
