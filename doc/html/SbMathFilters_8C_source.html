00035 double makeHouseholder(SBUpperMatrix* R, SBVector* Z, SBMatrix* AObs, SBVector* ZObs, unsigned int NumOfData)
00036 {
00037   unsigned int          Nx = R->n();
00038   unsigned int          i,j,l;
00039   double                s,ul,g;
00040 
00041   SBVector*             U = new SBVector(AObs->nRow(),"Aux Vector U");
00042 
00043   U->clear();                                                   // U-vector, + ul -- the first elementh of U
00044   for (l=0; l<Nx; l++)
00045     {
00046       s = R->at(l,l)*R->at(l,l);
00047       for (i=0; i<NumOfData; i++)
00048         s+=AObs->at(i,l)*AObs->at(i,l);
00049       if (s>0.0)
00050         {
00051           s = -sign(R->at(l,l))*sqrt(s);
00052           ul = R->at(l,l) - s;                                  // U_l
00053           for (i=0; i<NumOfData; i++) U->set(i, AObs->at(i,l)); // U_i (i=l+1, N)
00054           R->set(l,l, s);
00055           s = 1.0/(s*ul);                                       // betta
00056           for (j=l+1; j<Nx; j++)                                // |<-- processing "Ry/Ay"-block
00057             {                                                   // |
00058               g = ul*R->at(l,j);                                // |
00059               for (i=0; i<NumOfData; i++) g+=U->at(i)*AObs->at(i,j);    // |
00060               (*R)(l,j)+=s*g*ul;                                // |
00061               for (i=0; i<NumOfData; i++) (*AObs)(i,j)+=s*g*U->at(i);   // |
00062             };                                                  // |_________________________
00063           g = ul*Z->at(l);                                      // |--
00064           for (i=0; i<NumOfData; i++) g+=U->at(i)*ZObs->at(i);  // | dealing with the vectors Zy
00065           (*Z)(l)+=s*g*ul;                                      // |            and Z
00066           for (i=0; i<NumOfData; i++) (*ZObs)(i)+=s*g*U->at(i); // |--
00067         }
00068       else
00069         std::cerr << "got a zero during Householdering at l[" + QString().setNum(l) + "]\n";
00070     };
00071   
00072   s = 0.0;
00073   for (i=0; i<NumOfData; i++)
00074     s+= ZObs->at(i)*ZObs->at(i);
00075   return s;
00076 };

00330 void SBFilterModel::analyseSeries()
00331 {
00332   if (!IsOK)
00333     {
00334       Log->write(SBLog::DBG, SBLog::DATA, ClassName() + 
00335                  ": cannot make analysis of the series: !IsOK");
00336       return;
00337     };
00338 
00339   // collect derivatives:
00340   SBVector              *ZObs = new SBVector(Series->count(), "Vector of Obs");
00341   SBMatrix              *AObs = new SBMatrix(Series->count(), numOfUnknowns(), "Matrix of Derivatives");
00342   SBDataSeriesEntry     *E=NULL;
00343   unsigned int          i,j;
00344   double                a, fSig, t;
00345 
00346   SumW2 = 0.0;
00347   i = j = 0;
00348   for (E=Series->first(); E; i++, E=Series->next())
00349     if (DataAttribute->at(i)>0.5)
00350       {
00351         fSig = IdxSig?E->data()->at(IdxSig):1.0;
00352         t    = E->t();
00353         a = 1.0;
00354         for (i=0; i<NumOfPolinoms; i++)
00355           {
00356             AObs->set(j, i, a/fSig);
00357             a*=(t - T0)/365.25;
00358           };
00359         for (i=0; i<NumOfHarmonics; i++)
00360           {
00361             AObs->set(j, NumOfPolinoms + 2*i,   cos(2.0*M_PI*(t-T0)/Periods->at(i))/fSig);
00362             AObs->set(j, NumOfPolinoms + 2*i+1, sin(2.0*M_PI*(t-T0)/Periods->at(i))/fSig);
00363           };
00364         if (IsRunSM)
00365           {
00366             a = 1.0;
00367             for (i=0; i<NumOfSMTerms; i++)
00368               {
00369                 AObs->set(j, NumOfPolinoms + 2*NumOfHarmonics + 2*i,   a*cos(2.0*M_PI*(t-T0)/SMPeriod)/fSig);
00370                 AObs->set(j, NumOfPolinoms + 2*NumOfHarmonics + 2*i+1, a*sin(2.0*M_PI*(t-T0)/SMPeriod)/fSig);
00371                 a*=(t - T0)/365.25;
00372               };
00373           };
00374         
00375         ZObs->set(j, E->data()->at(Idx)/fSig);
00376         SumW2 += 1.0/fSig/fSig;
00377         j++;
00378       };
00379   NumOfData = j;
00380   
00381   // solve equations system:
00382   SBVector      *Z = new SBVector(numOfUnknowns(),  "Vector Z");
00383   SBUpperMatrix *R = new SBUpperMatrix(numOfUnknowns(),   "Matrix R");
00384   SBVector      *X = new SBVector(numOfUnknowns(), "Vector of Unknowns");

00386 
00387   for (i=0; i<numOfUnknowns(); i++)
00388     R->set(i,i, 1.0/1000000000.0);
00389  
00390   VTPV = makeHouseholder(R, Z, AObs, ZObs, NumOfData);
00391   Solve(*R, *X, *Z);
00392   RRT(*P, ~*R);
00393  
00394   // fill the vectors of parameters and std.vars:
00395   fSig = VTPV/(NumOfData - numOfUnknowns());
00396   for (i=0; i<NumOfPolinoms; i++)
00397     {
00398       Polinoms          -> set(i, X->at(i));
00399       PolinomsSigma     -> set(i, sqrt(fSig*P->at(i,i)));
00400       Log->write(SBLog::DBG, SBLog::DATA, ClassName() + ": data column #" + 
00401                  QString("").sprintf("(%d,%d): ", Idx, IdxSig) +
00402                  QString("").sprintf("Polinom order: %d; Value: %11.8f (%10.8f)", 
00403                                      i, Polinoms->at(i), PolinomsSigma->at(i)));
00404     };
00405   for (i=0; i<NumOfHarmonics; i++)
00406     {
00407       Harmonics         -> set(i,0, X->at(NumOfPolinoms + 2*i));
00408       HarmonicsSigma    -> set(i,0, sqrt(fSig*P->at(NumOfPolinoms + 2*i, NumOfPolinoms + 2*i)));
00409       Harmonics         -> set(i,1, X->at(NumOfPolinoms + 2*i+1));

00411       Log->write(SBLog::DBG, SBLog::DATA, ClassName() + ": data column #" + 
00412                  QString("").sprintf("(%d,%d): ", Idx, IdxSig) +
00413                  //              QString("").sprintf("Harmonics# %d (P=%10.4f); Cos: %11.8f (%10.8f), Sin: %11.8f (%10.8f)", 
00414                  QString("").sprintf("Harmonics# %d (P=%10.4f); Cos: %8.4f (%7.4f), Sin: %8.4f (%7.4f)", 
00415                                      i, Periods->at(i),
