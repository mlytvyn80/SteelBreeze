00194 Vector3 SBPlateEntry::velocity(const Vector3& R) const
00195 {
00196   Vector3       D(v3Zero);
00197 
00198   D[X_AXIS] = Omega[Y_AXIS]*R.at(Z_AXIS) - Omega[Z_AXIS]*R.at(Y_AXIS);
00199   D[Y_AXIS] = Omega[Z_AXIS]*R.at(X_AXIS) - Omega[X_AXIS]*R.at(Z_AXIS);
00200   D[Z_AXIS] = Omega[X_AXIS]*R.at(Y_AXIS) - Omega[Y_AXIS]*R.at(X_AXIS);
00201 
00202   return D*1.0e-9; //nrad to rad

01144     curvilinear geodetic coordinates (lambda, fi, h) reffered to an ellipsoid of semi-major 
01145     axis a and flattening f, using the following code (Borkowski, 1989). First compute 
01146     lambda=tan^(-1)(y/x) properly determining the quadrant from x and y (0<=lamda<=2pi)
01147   */
01148 
01149   Lam = atan2(R_[Y_AXIS], R_[X_AXIS]);
01150   if (Lam<0.0) Lam+=M_PI*2.0;
01151   
01152   /* 
01153      and r=sqrt(x^2+y^2).
01154   */
01155 
01156   double r = hypot(R_[X_AXIS], R_[Y_AXIS]);
01157 
01158   /*
01159         subroutine GEOD(r,z,fi,h)
01160     c Program to transform Cartesian to geodetic coordinates
01161     c based on the exact solution (Borkowski, 1989)
01162     c Input : r, z = equatorial [m] and polar [m] components
01163     c Output: fi, h = geodetic coord's (latitude [rad], height [m])
01164         implicit real*8(a-h,o-z)
01165   */
01166   
01167   double b,E,F,P,Q,D,s,v,G,t;
01168 
01169   /*
01170     c GRS80 ellipsoid: semimajor axis (a) and inverse flattening (fr)
01171         data a,fr /6378137.d0,298.257222101d0/
01172   */
01173   const double a  = 6378137.0;
01174   const double fr = 298.257222101;
01175 
01176   /*
01177         b = dsign(a - a/fr,z)
01178         E = ((z + b)*b/a - a)/r
01179         F = ((z - b)*b/a + a)/r
01180   */
01181 
01182   b = sign(a - a/fr, R_[Z_AXIS]);
01183   E = ((R_[Z_AXIS] + b)*b/a - a)/r;
01184   F = ((R_[Z_AXIS] - b)*b/a + a)/r;
01185     
01186   /*
01187     c Find solution to:  t**4 + 2*E*t**3 + 2*F*t - 1 = 0
01188         P = (E*F + 1.)*4.d0/3.d0
01189         Q = (E*E - F*F)*2.d0
01190         D = P*P*P + Q*Q
01191             if(D.ge.0d0) then
01192         s = dsqrt(D) + Q
01193         s = dsign(dexp(dlog(dabs(s))/3.d0),s)
01194         v = P/s - s
01195     c Improve the accuracy of numeric values of v
01196         v = -(Q + Q + v*v*v)/(3.d0*P)
01197             else
01198         v = 2.d0*dsqrt(-P)*dcos(dacos(Q/P/dsqrt(-P))/3.d0)
01199             endif
01200         G = .5d0*(E + dsqrt(E*E + v))
01201         t = dsqrt(G*G + (F - v*G)/(G + G - E)) - G
01202   */
01203 
01204   P = (E*F + 1.0)*4.0/3.0;
01205   Q = (E*E - F*F)*2.0;
01206   D = P*P*P + Q*Q;
01207   if (D>=0.0)
01208     {
01209       s = sqrt(D) + Q;
01210       s = sign(exp(log(fabs(s))/3.0), s);
01211       v = P/s - s;
01212       v = -(Q + Q + v*v*v)/(3.0*P);
01213     }
01214   else
01215     v = 2.0*sqrt(-P)*cos(acos(Q/P/sqrt(-P))/3.0);
01216 
01217   G = 0.5*(E + sqrt(E*E + v));
01218   t = sqrt(G*G + (F - v*G)/(G + G - E)) - G;
01219   
01220   /*
01221         fi = datan((1.d0 - t*t)*a/(2.d0*b*t))
01222         h = (r - a*t)*dcos(fi) + (z - b)*dsin(fi)
01223         return
01224         end
01225   */
01226   
01227   Fi = atan((1.0 - t*t)*a/(2.0*b*t));
01228   Ro = (r - a*t)*cos(Fi) + (R_[Z_AXIS] - b)*sin(Fi);
01229   return;
01230 };
