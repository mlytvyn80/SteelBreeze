00139 const Matrix3& RotMatrix::operator()(double Angle_)
00140 {
00141   //  sincosl(Angle=Angle_, &sa, &ca);
00142   Angle=Angle_;
00143   sa = sinl(Angle);
00144   ca = cosl(Angle);
00145   switch (Axis)
00146     {
00147     case X_AXIS:
00148       mat[0][0]= 1.0;   mat[0][1]= 0.0; mat[0][2]= 0.0;
00149       mat[1][0]= 0.0;   mat[1][1]= ca;  mat[1][2]= sa;
00150       mat[2][0]= 0.0;   mat[2][1]=-sa;  mat[2][2]= ca;
00151       break;
00152     case Y_AXIS:
00153       mat[0][0]= ca;    mat[0][1]= 0.0; mat[0][2]=-sa;
00154       mat[1][0]= 0.0;   mat[1][1]= 1.0; mat[1][2]= 0.0;
00155       mat[2][0]= sa;    mat[2][1]= 0.0; mat[2][2]= ca;
00156       break;
00157     case Z_AXIS:
00158       mat[0][0]= ca;    mat[0][1]= sa;  mat[0][2]= 0.0;
00159       mat[1][0]=-sa;    mat[1][1]= ca;  mat[1][2]= 0.0;
00160       mat[2][0]= 0.0;   mat[2][1]= 0.0; mat[2][2]= 1.0;
00161       break;
00162     };
