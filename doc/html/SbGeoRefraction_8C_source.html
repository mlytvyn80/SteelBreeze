00403                              const SBStation *Station, const SBMJD& MJD)
00404 {
00405   double a_dry_avg, a_dry_amp, b_dry_avg, b_dry_amp, c_dry_avg, c_dry_amp;
00406 
00407   double phi  = Station->latitude();
00408 
00416   double sinE = sin(E);
00417   double a = 2.53e-5;
00418   double b = 5.49e-3;
00419   double c = 1.14e-3;
00420   double deltaM = (1.0/sinE - (1.0 + a/(1.0 + b/(1.0 + c)))/(sinE + a/(sinE + b/(sinE + c))))*
00421     Station->height()/1000.0;
00422   
00423   if (fabs(phi)<= ModelArg[0])
00424     {
00425       a_dry_avg = ModelA_avg[0];
00426       a_dry_amp = ModelA_amp[0];
00427       b_dry_avg = ModelB_avg[0];
00428       b_dry_amp = ModelB_amp[0];
00429       c_dry_avg = ModelC_avg[0];
00430       c_dry_amp = ModelC_amp[0];
00431     }
00432   else if (fabs(phi)>= ModelArg[4])
00433     {
00434       a_dry_avg = ModelA_avg[4];
00435       a_dry_amp = ModelA_amp[4];
00436       b_dry_avg = ModelB_avg[4];
00437       b_dry_amp = ModelB_amp[4];
00438       c_dry_avg = ModelC_avg[4];
00439       c_dry_amp = ModelC_amp[4];
00440     }
00441   else
00442     {
00443       int i = 0;
00444       while (++i<4 && fabs(phi)>ModelArg[i]) ;
00445 
00446       a_dry_avg = linterpol(ModelArg[i-1], ModelArg[i], ModelA_avg[i-1], ModelA_avg[i], phi);

00459 
00460   return (1.0 + a/(1.0 + b/(1.0 + c)))/(sinE + a/(sinE + b/(sinE + c))) + deltaM;

00684 const Vector3 SBRefraction::refrDir(float T, double DryZD, double WetZD, 
00685                                     const SBStation* Station, const SBSource* Source, const Vector3& S)
00686 {
00687   Vector3 s = ~Station->fmVEN()*S;      // s is abberated source in VEN, TRS
00688   Azimuth   = atan2(s[EAST], s[NORT]);
00689   Delta     = 8567.0/292.0*(T + 273.15);
00690   Elev = RefrElev = M_PI/2.0;
00691   if (s.at(VERT)<0.0)
00692     Log->write(SBLog::ERR, SBLog::REFRACT, ClassName() + ": ERROR: source [" + Source->name() + 
00693                "] below the horizon at the station [" + Station->name() + "]");
00694   if (s[VERT]!=1.0)
00695     {
00696       Elev         = asin(s[VERT]);
00697       double sinE2 = s[VERT]*s[VERT];           // =sin(E0)*sin(E0)
00698       //M_001 has been calculated with the follow mean values: 
00699       //                        \alpha=5, \betta=3.5, q_1=0.1459, q_2=1.424
00700       //      double Mw    = 0.29602606;        
00701       double Sigma = Delta/6378136.49;
00702       double Sigma_= sinE2*(sqrt(1.0 + Sigma*(Sigma + 2)/sinE2) - 1.0);
00703       double Aleph = acos((1.0 + Sigma_)/(1.0 + Sigma));
00704       double Hi    = (DryZD + WetZD/*/Mw*/)/Delta;
00705       //      dE           = acos(cos(Elevation + Aleph)/(1.0 + Hi)) - Aleph;
00706       RefrElev = acos(cos(Elev + Aleph)/(1.0 + Hi)) - Aleph;
00707       dE = RefrElev - Elev;
00708       double sinE, cosE, sinT, cosT;
00709       sincos(RefrElev, &sinE, &cosE);   //Elevation here is E0+dE
00710       sincos(Azimuth, &sinT, &cosT);
00711       // set up s with azimuth and corrected for refraction elevation (VEN)
00712       s[VERT] = sinE;                           // vertical
00713       s[EAST] = cosE*sinT;                      // east
00714       s[NORT] = cosE*cosT;                      // north
00715     };
00716   s = Station->fmVEN()*s; // turn back to TRF
00717   return s;
00718 };

00728 double SBRefraction::altitudeCorr(float /*T*/, double DryZD, double StarTime,
00729                                   const SBStation* Station, const SBSource* Source)
00730 {
00731   double sinE, cosE;
00732   double Psi = 0.0;
00733 
00734   if (Station->mountType()==SBStation::Mnt_AZEL || Station->axisOffset()==0.0)
00735     return 0.0;
00736   if (Station->mountType()==SBStation::Mnt_Unkn)
00737     return 0.0;
00738 
00739 
00740   sincos (refrElev(), &sinE, &cosE);
00741   switch (Station->mountType())
00742     {
00743     case SBStation::Mnt_EQUA:
00744       Psi = cos(Station->latitude())*cos(StarTime + Station->longitude() - Source->ra_t());
00745       break;
00746     case SBStation::Mnt_X_YN:
00747       {
00748         double cosT = cos(azimuth());
00749         Psi = sinE/sqrt(1.0 - cosT*cosT*cosE*cosE);
00750       };
00751       break;
00752     case SBStation::Mnt_X_YE:
00753       {
00754         double sinT = sin(azimuth());
00755         Psi = sinE/sqrt(1.0 - sinT*sinT*cosE*cosE);
00756       };
00757       break;
00758     case SBStation::Mnt_Richmond:
00759       Psi = cos(39.023833*DEG2RAD)*
00760         cos(atan(cosE*sin(azimuth() - 0.12*DEG2RAD)/
00761              (cos(39.023833*DEG2RAD)*sinE - sin(39.023833*DEG2RAD)*cosE*cos(azimuth() + 
00762                                                                             0.12*DEG2RAD))));
00763       break;
00764     default:
00765       break;
00766     };
00767   return -DryZD*Station->axisOffset()/Delta*Psi;
00768 };
