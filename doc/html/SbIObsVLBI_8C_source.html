02084 void SBVLBISetView::import(SBVLBISession::ObsImportedFrom IType)
02085 {
02086   if (numOpenedWindows) return; //there are opened windows dependent upon us..
02087 
02088   bool          IsNeedDump = CBDump->isChecked();
02089   int           ExistedSessMode;
02090   bool          IsLoaded   = FALSE;
02091   QFileInfo     FInfo;
02092 
02093   ExistedSessMode = 0;
02094   if (RBIfExistedImport->isChecked())
02095     ExistedSessMode = -1;
02096   else if (RBIfExistedDrop->isChecked())
02097     ExistedSessMode = 1;
02098 
02099   Wvlbi->loadMaster();
02100   qApp->processEvents();
02101   QStringList sl(QFileDialog::getOpenFileNames(0, SetUp->path2Compil()));
02102   numOpenedWindows++;
02103   if (sl.count())
02104     {
02105       emit stBarReset();
02106       emit stBarSetTotalSteps(sl.count());
02107       int i=1;
02108       for (QStringList::Iterator it=sl.begin(); it != sl.end(); ++it)
02109         { 
02110           bool isIgnore=FALSE;
02111           emit message(QString("Processing: ") + *it);
02112           qApp->processEvents();
02113           FInfo.setFile(*it);
02114           
02115           if (FInfo.size()<100)
02116             {
02117               Log->write(SBLog::DBG, SBLog::INTERFACE | SBLog::IO | SBLog::DATA, 
02118                          ClassName() + ": file [" + *it + "] is suspiciously small; ignored");

02123               Log->write(SBLog::DBG, SBLog::INTERFACE | SBLog::IO | SBLog::DATA, 
02124                          ClassName() + ": file [" + *it + "] has \".bad\" extension; ignored");
02125               isIgnore=TRUE;
02126             };

02129               FInfo.setFile(FInfo.readLink());

02133                              ClassName() + ": file [" + *it + 
02134                              "] points to the file with \".bad\" extension; ignored");
02135                   isIgnore=TRUE;
02136                 };
02137             };
02138           if (!isIgnore && Wvlbi->import(IType, *it, IsNeedDump, ExistedSessMode))
02139             {
02140               IsLoaded = TRUE;
02141               drawInfo();
02142             };
02143           emit stBarSetProgress(i++);
02144         };
02145       drawInfo();
02146       emit stBarReset();
02147       qApp->processEvents();
02148     }
02149   Wvlbi->releaseMaster();
02150   if (IsLoaded)
02151     {
02152       isModified = TRUE;
