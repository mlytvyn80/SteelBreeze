00229 void SBTropRecord::collect(SBTropRecord* Tr)
00230 {
00231   if (!IsCollecting)
00232     {
00233       clear();
00234       IsCollecting = TRUE;
00235     };
00236 
00237   if (TSum == 0.0)
00238     TSum = Tr->NumPts**(SBMJD*)Tr;
00239   else
00240     TSum += Tr->NumPts**(SBMJD*)Tr;
00241 
00242 
00243   double p0,p1,p2;
00244 
00245   p0 = Tr->WetZenithErr*Tr->WetZenithErr;
00246   p1 = Tr->GradNorthErr*Tr->GradNorthErr;
00247   p2 = Tr->GradEastErr *Tr->GradEastErr;
00248   
00249   aleph[0] += 1.0/p0;   beta[0] += Tr->WetZenith/p0;    gamma[0] += Tr->WetZenith*Tr->WetZenith/p0;
00250   aleph[1] += 1.0/p1;   beta[1] += Tr->GradNorth/p1;    gamma[1] += Tr->GradNorth*Tr->GradNorth/p1;
00251   aleph[2] += 1.0/p2;   beta[2] += Tr->GradEast /p2;    gamma[2] += Tr->GradEast *Tr->GradEast /p2;
00252 
00253   delta[0] += p0;
00254   delta[1] += p1;
00255   delta[2] += p2;
00256 
00257   TotZenith+= Tr->TotZenith/p0;
00258   NumPts   += Tr->NumPts;
00259   Temp     += Tr->Temp;
00260   Pres     += Tr->Pres;
00261   RH       += Tr->RH;
00262   NumProc++;

00265 void SBTropRecord::processCollected()

00267   if (!IsCollecting || !NumPts)
00268     return;

00291       //      WetZenithErr = sqrt((gamma[0] - beta[0]*beta[0]/aleph[0])/aleph[0]);
00292       //      WetZenithErr = sqrt(delta[0]/NumProc);
