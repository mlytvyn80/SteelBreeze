00346 /*                                                                                              */
00347 /* class SBDBHDRecT<Class>                                                                      */
00348 /* Data record: N x Class (all other kinds of input data)                                       */
00349 /*==============================================================================================*/
00350 template<class Class> class SBDBHDRecT : public SBDBHPhysRec
00351 {
00352 protected:
00353   Class*        nD;
00354   int           Num;
00355 public:
00356   SBDBHDRecT(){nD=NULL; Num=0;};
00357   virtual ~SBDBHDRecT(){if (nD) delete []nD;};
00358   virtual Class operator[](int i){return nD&&-1<i&&i<Num?*(nD+i):(Class)0;};
00359   virtual Class val(SBDBHPar* Par, int dim1, int dim2, int dim3) // zero based indexes
00360     {
00361       return (dim1<Par->dim1()&&dim2<Par->dim2()&&dim3<Par->dim3()&&dim1>=0&&dim2>=0&&dim3>=0)?
00362         operator[](Par->offset() + dim1 + Par->dim1()*dim2 + Par->dim1()*Par->dim2()*dim3)
00363         :(Class)0;
00364     };
00365   int num(){return Num;};
00366   virtual int parseLR(SBDS_dbh& s)
00367     {
00368       if (nD) delete []nD;
00369       nD=new Class[(Num=Length/sizeof(Class))];  
00370       Class *p=nD; 
00371       for(int i=0; i<Num; i++,p++) s>>*p;
00372       return Length;
00373     };
00374 };
