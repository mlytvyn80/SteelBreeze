00773 void SBObsVLBIEntry::process(SBRunManager* Mgr)
00774 {
00775   SBDelay       *CalcedDelay = Mgr->delay();
00776   const SB_TRF  *TRF      = Mgr->trf();
00777   const SB_CRF  *CRF      = Mgr->crf();
00778   //
00779   // station #1:
00780   SBStationInfo *St_1Info = NULL;
00781   SBStation     *St_1     = NULL;
00782   // station #2:
00783   SBStation     *St_2     = NULL;
00784   SBStationInfo *St_2Info = NULL;
00785   // source:
00786   SBSource      *Src      = NULL;
00787   SBSourceInfo  *SrInfo   = NULL;
00788   // baseline:
00789   SBBaseInfo    *BaseInfo = NULL;
00790   //
00791   double        O_C = 0.0;
00792   double        DelayAprS = 15.0e-14;
00793   double        DelayS    = DelayErr>0.0?DelayErr:200.0e-12;
00794   double        DelayIonS = DelayIonErr>0.0?DelayIonErr:200.0e-12;
00795 
00796   double        DelaySigma= sqrt(DelayS*DelayS + DelayIonS*DelayIonS + DelayAprS*DelayAprS);
00797   double        Weight    = 1.0/DelaySigma;
00798   
00799   
00800   if (!(St_1=station1(TRF, St_1Info)))
00801     Log->write(SBLog::ERR, SBLog::STATION | SBLog::DELAYALL, 
00802                ClassName() + ": cannot find the station #1 [" + (St_1Info?St_1Info->toString():"NULL") + 
00803                "] aka [" + (St_1Info?St_1Info->aka():"NULL") + "] in the catalogue");
00804   else if (!(St_2=station2(TRF, St_2Info)))
00805     Log->write(SBLog::ERR, SBLog::STATION | SBLog::DELAYALL, 
00806                ClassName() + ": cannot find the station #2 [" + (St_2Info?St_2Info->toString():"NULL") + 
00807                "] aka [" + (St_2Info?St_2Info->aka():"NULL") + "] in the catalogue");
00808   else if (!(Src=source(CRF, SrInfo)))
00809     Log->write(SBLog::ERR, SBLog::SOURCE | SBLog::DELAYALL, 
00810                ClassName() + ": cannot find the source [" + (SrInfo?SrInfo->name():"NULL") + 
00811                "] aka [" + (SrInfo?SrInfo->aka():"NULL") + "] in the catalogue");
00812   else 
00813     {
00814       St_1->calcDisplacement(Mgr, *this);
00815       St_2->calcDisplacement(Mgr, *this);
00816       Src ->updateParameters(Mgr, *this);
00817 
00818       if (isAttr(breakClock1) && (Mgr->prj()->cfg().p().clock0().type()==SBParameterCfg::PT_STH))
00819         St_1->p_C0()->addAttr(SBDerivation::IsBreak);
00820       if (isAttr(breakClock2) && (Mgr->prj()->cfg().p().clock0().type()==SBParameterCfg::PT_STH))
00821         St_2->p_C0()->addAttr(SBDerivation::IsBreak);
00822       
00823       double    dt = (*this - Owner->tMean());
00824       double    c1=0.0, c2=0.0, d, Cabel1=0.9, Cabel2=0.0;
00825 
00826 
00827       d = 1.0;
00828       for (int i=0; i<St_1Info->cl_Order(); i++)
00829         {
00830           c1+=St_1->p_Clock(i)->v()*d;
00831           d*=dt;
00832         };
00833       d = 1.0;
00834       for (int i=0; i<St_2Info->cl_Order(); i++)
00835         {
00836           c2+=St_2->p_Clock(i)->v()*d;
00837           d*=dt;
00838         };
00839       
00840       Cabel1 = St_1Info->isAttr(SBStationInfo::BadCable)?0.0:
00841         St_1Info->isAttr(SBStationInfo::InvCable)?-Cable_1:Cable_1;
00842       Cabel2 = St_2Info->isAttr(SBStationInfo::BadCable)?0.0:
00843         St_2Info->isAttr(SBStationInfo::InvCable)?-Cable_2:Cable_2;
00844       if (St_1->p_Cable()->isAttr(SBDerivation::IsInRun))
00845         St_1->p_Cable()->setD( Cabel1, *this);
00846       if (St_2->p_Cable()->isAttr(SBDerivation::IsInRun))
00847         St_2->p_Cable()->setD(-Cabel2, *this);
00848 
00849       if (St_1->p_Cable()->v()!=0.0)
00850         Cabel1 *= 1.0 + St_1->p_Cable()->v();
00851 
00852       if (St_2->p_Cable()->v()!=0.0)
00853         Cabel2 *= 1.0 + St_2->p_Cable()->v();
00854 
00855       //      dCable = (St_2Info->isAttr(SBStationInfo::InvCable)?-Cable_2:Cable_2) - 
00856       //        (St_1Info->isAttr(SBStationInfo::InvCable)?-Cable_1:Cable_1);
00857       double delayC=0.0;
00858       /*
00859         O_C = DelayRes =
00860         Delay + (Cabel2 - Cabel1) + AmbiguityFactor*GrDelayAmbig 
00861         - ( 
00862         CalcedDelay->calc(Mgr, St_1, St_2, Src, this, St_1Info, St_2Info)
00863         + DelayIon 
00864         + (St_2Info->clockBreaks().bt(*this) - St_1Info->clockBreaks().bt(*this))*1.0e-12
00865         + (St_2Info->clocksValue(dt) - St_1Info->clocksValue(dt))
00866         + (c2 - c1)
00867         );
00868       */
00869       delayC = (Cabel1 - Cabel2)
00870         + CalcedDelay->calc(Mgr, St_1, St_2, Src, this, St_1Info, St_2Info)
00871         + DelayIon 
00872         + (St_2Info->clockBreaks().bt(*this) - St_1Info->clockBreaks().bt(*this))*1.0e-12
00873         + (St_2Info->clocksValue(dt) - St_1Info->clocksValue(dt))
00874         + (c2 - c1);
00875 
00876       O_C = DelayRes = Delay + AmbiguityFactor*GrDelayAmbig - delayC;
00877 
00878       
00879       /*=*/
00880       bool              IsTstOutput = TRUE;
00881       QString           StrTst;
00882       IsTstOutput = FALSE;
00883       if (IsTstOutput)
00884         {
00885           double        PhiCorr;
00886           double        delTau = delayC;
00887 
00888           delTau = (Cabel1 - Cabel2)
00889             + CalcedDelay->totalDelay()
00890             - DelayIon 
00891             + (St_2Info->clockBreaks().bt(*this) - St_1Info->clockBreaks().bt(*this))*1.0e-12
00892             + (St_2Info->clocksValue(dt) - St_1Info->clocksValue(dt))
00893             + (c2 - c1);
00894           
00895           PhiCorr = 360.0*fmod(refFreq()*1.0e6*delTau, 1.0);
00896           StrTst.sprintf("%11.6f  [%-8s]-[%-8s] %8s", 
00897                          PhiCorr,
00898                          (const char*)St_1Info->aka(), 
00899                          (const char*)St_2Info->aka(), 
00900                          (const char*)SrInfo->name()
00901                          );
00902           std::cout << StrTst << "\n";
00903         };
00904       /*=*/
00905 
00906       /*
00907         static QString LastSessionName;
00908         static double ddd;
00909         
00910         if (LastSessionName != Owner->name())
00911         {
00912         LastSessionName = Owner->name();
00913         ddd = (rand()-RAND_MAX/2.0)/RAND_MAX;
00914         std::cout << LastSessionName << "\n";
00915         };
00916         
00917         O_C = DelayRes = 
00918         1.0e-9*ddd*(station_1Idx() + station_2Idx());
00919         //            (Cabel2 - Cabel1);
00920       */
00921 
00922       
00923       ZenithDelay_1 = CalcedDelay->totZenithDelay1();
00924       ZenithDelay_2 = CalcedDelay->totZenithDelay2();
00925       
00926       d = 1.0;
00927       int nCl = Mgr->prj()->cfg().p().clock0().type()==SBParameterCfg::PT_STH?1:
00928         std::max(St_1Info->cl_Order(), St_2Info->cl_Order());
00929       for (int i=0; i<nCl; i++)
00930         {
00931           if (St_1->p_Clock(i)->isAttr(SBDerivation::IsInRun))
00932             St_1->p_Clock(i)->setD(-d, *this);
00933           if (St_2->p_Clock(i)->isAttr(SBDerivation::IsInRun))
00934             St_2->p_Clock(i)->setD( d, *this);
00935           d*=dt;
00936         };
00937       
00938       SBNamed            baseName(St_1Info->toString() + ":" + St_2Info->toString());

00957                    ClassName() + ": cannot find the station_1 [" + St_1->id().toString() + "] aka [" +

01770 SBObsVLBIEntry* SBVLBISession::parseObsDumpString(const QString& Str, SBMJD& t, 
01771                                                   SBStationInfo*& St_1, SBStationInfo*& St_2, SBSourceInfo*& So)
01772 {
01773   bool                  Re=TRUE;
01774   QString               ss;
01775   SBStationID           id;
01776   SBObsVLBIEntry*       E = NULL;
01777   int                   idx, i;
01778   //          1         2         3         4         5
01779   //012345678901234567890123456789012345678901234567890123456789
01780   // +SCAN: 523 044091:26340.0000 40440S002 40439S002 1642+690
01781   //           012345678901234567890123456789012345678901234567890123456789
01782   
01783   // idx:
01784   i = Str.find(" ", 8);
01785   idx = Str.mid(8,i-8).toInt(&Re);
01786   if (!Re || idx<0)
01787     {
01788       Log->write(SBLog::WRN, SBLog::DATA, ClassName() + ": wrong format of UI string (SCAN::idx)");
01789       return NULL;
01790     };
01791 
01792   // time:
01793   t.setMJD(SBMJD::F_INTERNAL, Str.mid(i+1, 17));
01794   if (!t.isGood())
01795     {
01796       Log->write(SBLog::WRN, SBLog::DATA, ClassName() + ": wrong format of UI string (SCAN::t)");
01797       return NULL;
01798     };
01799   
01800   // Station_1:
01801   if (SBStationID::isValidStr((ss = Str.mid(i+19, 9))))
01802     {
01803       id.setID(ss);
01804       if (!(St_1 = StationList.find(&id)))
01805         {
01806           Log->write(SBLog::WRN, SBLog::DATA, ClassName() + ": cannot find station [" + ss + "] in the list");
01807           return NULL;
01808         };
01809     }
01810   else
01811     {
01812       Log->write(SBLog::WRN, SBLog::DATA, ClassName() + ": wrong format of UI string (SCAN::St_1)");
01813       return NULL;
01814     };
01815 
01816   // Station_2:
01817   if (SBStationID::isValidStr((ss = Str.mid(i+29, 9))))
01818     {
01819       id.setID(ss);
01820       if (!(St_2 = StationList.find(&id)))
01821         {
01822           Log->write(SBLog::WRN, SBLog::DATA, ClassName() + ": cannot find station [" + ss + "] in the list");
01823           return NULL;
01824         };
01825     }
01826   else
01827     {
01828       Log->write(SBLog::WRN, SBLog::DATA, ClassName() + ": wrong format of UI string (SCAN::St_2)");
01829       return NULL;
01830     };
01831 
01832   // Source:
01833   SBNamed       N((ss = Str.mid(i+39)));
01834   if (!(So = SourceList.find(&N)))
01835     {
01836       Log->write(SBLog::WRN, SBLog::DATA, ClassName() + ": cannot find source [" + ss + "] in the list");
01837       return NULL;
01838     };
01839   
01840   // guess by idx:
01841   if (idx<(int)count() && (E=at(idx)) && (fabsl(*E-t)<0.05/86400.0) && 
01842       E->station_1Idx()==St_1->idx() && 
01843       E->station_2Idx()==St_2->idx() && 
01844       E->sourceIdx()==So->idx())
01845     return E;
01846   else // if failed, look for proper combination:
01847     {
01848       E=first();
01849       while (E && (fabsl(*E-t)<0.05/86400.0) && 
01850              E->station_1Idx()==St_1->idx() && 
01851              E->station_2Idx()==St_2->idx() && 
01852              E->sourceIdx()==So->idx())
01853         E=next();
01854     };
01855   
01856   return E;
01857 };
