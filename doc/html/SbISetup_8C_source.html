00594 void SBSetupDialog::addInst()
00595 {
00596   QString Acronym_ = leAcronym->text();
00597   QString Descr_ = leDescr->text();
00598   if (Acronym_.simplifyWhiteSpace()!="" && Descr_.simplifyWhiteSpace()!="")
00599     {
00600       if (Acronym_.length()<6)
00601         if (InstList->find(Acronym_))
00602           QMessageBox::warning(this, "Organization exist already", 
00603                                "The acronym you typed already exist\n"
00604                                "in the list. Can't insert a duplicate.", "OK");
00605         else
00606           {
00607             SBInstitution* Item=new SBInstitution(Acronym_, Descr_);
00608             if (InstList->insert(Item))
00609               {
00610                 (void) new SBInstLI(lvInst, Item);
00611                 leAcronym->setText("");
00612                 leDescr->setText("");
00613               }
00614             else delete Item;
00615           }
