00048 QTextStream &operator>>(QTextStream& s, SBDataSeriesEntry& entry)
00049 {
00050   QString       str;
00051   bool          IsOK;
00052   int           i, j, l, m, idx;
00053   double        d;
00054   int           NumOfFields=0;
00055 
00056   str = s.readLine().simplifyWhiteSpace();
00057   if (!(l=str.length()))
00058     return s;
00059 
00060   if (str.mid(0,1)=="#")        // it's a comment
00061     return s;
00062 
00063   // determine the number of fields:
00064   i=0;
00065   j=0;
00066   while (i<l)
00067     {
00068       j=str.find(" ", i+1);
00069       if (j!=-1)
00070         {
00071           NumOfFields++;
00072           i=j;
00073         }
00074       else 
00075         i=l;
00076     };
00077   
00078   if (!NumOfFields)     // only one field
00079     return s;
00080   
00081   // create the data records:
00082   entry.Data = new SBVector(NumOfFields, "data record");
00083 
00084   // get the argument:
00085   j=str.find(" ", 0);
00086   d=str.mid(0, j).toDouble(&IsOK);
00087   if (!IsOK)            // arg is not a double
00088     return s;
00089   entry.T=d;
00090   
00091   // get the fields of data:
00092   i=j;
00093   idx=0;
00094   m=0;
00095   while ((i!=-1) && IsOK)
00096     {
00097       j=str.find(" ", i+1);
00098       if (j>i)
00099         {
00100           d = str.mid(i+1,j-i-1).toDouble(&IsOK);
00101           if (IsOK)
00102             {
00103               entry.Data->set(idx, d);
00104               idx++;
00105             };
00106           m++;
00107         }
00108       else if (m+1==NumOfFields) // last field
00109         {
00110           d = str.mid(i+1).toDouble(&IsOK);
00111           if (IsOK)
00112             {
00113               entry.Data->set(idx, d);
00114               idx++;
00115             };
00116         };
00117       i=j;
00118     };
00119 
00120   if (IsOK)
00121     entry.IsReadOK = TRUE;
00122 
00123   return s;

00148 void SBDataSeries::checkRecords()
00149 {
00150   int           Num;
00151 
00152   DT    = 0.0;
00153   DTMin = 0.0;
00154   DTMax = 0.0;
00155   IsChecked = FALSE;
00156   NumOfDataColumns = 0;
00157 
00158   if ((Num=count()))
00159     {
00160       unsigned int      idx=0;
00161       unsigned int      NoC=0, MaxNoC=0, MinNoC=0, n;
00162       double            DT_sum=0.0, DT_tmp=0.0;
00163       double            TPrev =0.0;
00164 
00165       sort();
00166       IsEquiDistant = TRUE;
00167 
00168       for (SBDataSeriesEntry *E=first(); E; E=next(), idx++)
00169         {
00170           if (idx)
00171             {
00172               DT_sum+= (DT_tmp = E->t() - TPrev);
00173               if (DT==0.0)
00174                 DT = DTMin = DTMax = DT_tmp;
00175               else if (DT!=DT_tmp)
00176                 {
00177                   if (DTMin > DT_tmp)
00178                     DTMin = DT_tmp;
00179                   if (DTMax < DT_tmp)
00180                     DTMax = DT_tmp;
00181                   IsEquiDistant = FALSE;
00182                 };
00183             }
00184           else
00185             NoC=MaxNoC=MinNoC = E->data()->n();
00186           TPrev = E->t();
00187           if ((n=E->data()->n()) != NoC)
00188             {
00189               if (n<MinNoC)
00190                 MinNoC = n;
00191               if (MaxNoC<n)
00192                 MaxNoC = n;
00193             };
00194         };
00195 
00196       if (!IsEquiDistant)
00197         DT = DT_sum/(Num-1);
00198       
00199       if (MaxNoC != MinNoC) // well, remove them..
00200         {
00201           int   NRm=0;
00202           for (idx=0; idx<count(); idx++)
00203             if (at(idx)->data()->n() != MaxNoC)
00204               {
00205                 if (remove(idx))
00206                   {
00207                     NRm++;
00208                     idx--;
00209                   }
00210                 else
00211                   Log->write(SBLog::ERR, SBLog::DATA, ClassName() + ": cannot remove the record #" + 
00212                              QString("_").setNum(idx) + " from the series");
00213               }
00214           Log->write(SBLog::DBG, SBLog::DATA, ClassName() + ": the series is cleaned, " +
00215                      QString("_").setNum(NRm) + " record freed");
00216         };
00217       
00218       NumOfDataColumns = MaxNoC;
00219       IsChecked = TRUE;
00220     };
00221 };
