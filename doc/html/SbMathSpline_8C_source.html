00082 void SBSpline::solveSpline(unsigned int nCol)
00083 {
00084   SBVector      Aleph(N, "Aleph");
00085   SBVector      Beta (N, "Beta");
00086   double        f;
00087   unsigned int  nF = 1+nCol, nM=1+NDimension+nCol;
00088 
00089   f = h(1) + h(2);
00090   Aleph(2) = -0.5*h(2)/f;
00091   Beta (2) = 3.0*(B->at(0, nF)/h(1) - B->at(1, nF)*(1.0/h(1) + 1.0/h(2)) + B->at(2, nF)/h(2))/f;
00092   
00093   for (unsigned int i=2; i<N-2; i++)
00094     {
00095       f = Aleph(i)*h(i) + 2.0*(h(i) + h(i+1));
00096       Aleph(i+1) = -h(i+1)/f;
00097       Beta (i+1) = (6.0*(B->at(i-1, nF)/h(i) - B->at(i, nF)*(1.0/h(i) + 1.0/h(i+1)) + B->at(i+1, nF)/h(i+1))
00098                     - Beta(i)*h(i))/f;
00099     };
00100 
00101   (*B)(N-1, nM) = 0.0;
00102   (*B)(N-2, nM) = (6.0*(B->at(N-3, nF)/h(N-2) - B->at(N-2, nF)*(1.0/h(N-2) + 1.0/h(N-1)) + 
00103                         B->at(N-1, nF)/h(N-1)) - h(N-2)*Beta(N-2)) /
00104     (h(N-2)*Aleph(N-2) + 2.0*(h(N-2) + h(N-1)));
00105 
00106   for (unsigned int i=N-3; i>0; i--)
00107       (*B)(i, nM) = Aleph.at(i+1)*B->at(i+1, nM) + Beta.at(i+1);
00108 
00109   (*B)(0, nM) = 0.0;
00110 };

00112 double SBSpline::spline(unsigned int nCol, double x)
00113 {
00114   if (!IsReady)
00115     {
00116       Log->write(SBLog::ERR, SBLog::SPLINE, ClassName() + ": the spline is not ready");
00117       return 0.0;
00118     };
00119   if (nCol>=NDimension)
00120     {
00121       Log->write(SBLog::ERR, SBLog::SPLINE, ClassName() + ": the nCol [" + QString().setNum(nCol) + 
00122                  "] out of range [0.." + QString().setNum(NDimension-1) + "]");
00123       return 0.0;
00124     };
00125   unsigned int  nF = 1+nCol, nM=1+NDimension+nCol, Idx = 0;
00126   
00127   // get the idx:
00128   if (x<=B->at(1, 0)) 
00129     Idx = 1;
00130   else if (B->at(N-1, 0)<=x)
00131     Idx = N-1;
00132   else
00133     {
00134       unsigned int i=2;
00135       while (i<N && B->at(i-1, 0)<=x) i++;
00136       Idx = i-1;
00137     };
00138   
00139   double f1 = B->at(Idx, 0) - x, f2 = x - B->at(Idx-1, 0), h2=h(Idx)*h(Idx);
00140   return (B->at(Idx-1, nM)*f1*f1*f1/6.0 + B->at(Idx, nM)*f2*f2*f2/6.0 +
00141           (B->at(Idx-1, nF) - B->at(Idx-1, nM)*h2/6.0)*f1 +
00142           (B->at(Idx,   nF) - B->at(Idx,   nM)*h2/6.0)*f2)/h(Idx);
00143 };
