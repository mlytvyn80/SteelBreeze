00324 void SBMEM::removeTrendsFromData()
00325 {
00326   double        A, B, C;
00327   double        Dr, Di, Er, Ei;
00328   
00329   double        sig2; // replace in the case of real weights
00330   double        t0 = X->at(0);
00331   double        f;
00332 
00333   sig2 = 1.0; // sig_i*sig_i
00334 
00335   A = B = C = Dr = Di = Er = Ei = 0.0;
00336   for (unsigned int i=0; i<N; i++)
00337     {
00338       A += 1.0/sig2;
00339       B += (f=(X->at(i) - t0))/sig2;
00340       C += f*f/sig2;
00341       Dr+= Z_r->at(i)/sig2;
00342       Di+= Z_i->at(i)/sig2;
00343       Er+= Z_r->at(i)*f/sig2;
00344       Ei+= Z_i->at(i)*f/sig2;
00345     };
00346   TrendR = (A*Er - B*Dr)/(A*C - B*B);
00347   ShiftR = (Dr - B*TrendR)/A;
00348   TrendI = (A*Ei - B*Di)/(A*C - B*B);
00349   ShiftI = (Di - B*TrendI)/A;
00350   for (unsigned int i=0; i<N; i++)
00351     {
00352       Z_r->set(i, Z_r->at(i) - ShiftR - TrendR*(X->at(i) - t0));
00353       Z_i->set(i, Z_i->at(i) - ShiftI - TrendI*(X->at(i) - t0));
00354     };  
00355 };

00395 void SBMEM::calcA_Recusion()
00396 {
00397   unsigned int                  i, m;
00398   double                        f;
00399   std::complex<double>          c(0.0, 0.0);
00400   std::complex<double>          *Save=NULL;
00401 
00402   // init: m=0
00403   //
00404   for (i=1; i<N; i++)
00405     *(EFm+i) = std::complex<double>(Z_r->at(i), Z_i->at(i));
00406   for (i=0; i<N-1; i++)
00407     *(EBm+i) = std::complex<double>(Z_r->at(i), Z_i->at(i));
00408   
00409   for (f=0.0, i=0; i<N; i++)
00410     f+=mod2(Z_r->at(i),Z_i->at(i));
00411   Pm = f/N;
00412   
00413   // a^1_1:
00414   for (c=CZero,i=0; i<N-1; i++)
00415     c += std::conj(*(EBm+i))**(EFm+i+1);
00416   for (f=0.0, i=0; i<N-1; i++)
00417     f += mod2(*(EBm+i)) + mod2(*(EFm+i+1));
00418   *Am = 2.0*c/f;
00419 
00420   // P_1:
00421   Pm_1 = Pm;
00422   Pm = Pm_1*(1.0 - mod2(*Am));
00423 
00424 
00425   // iteration: m=1,M-1
00426   for (m=1; m<M; m++)
00427     {
00428       // swap pointers:
00429       // Am:
00430       Save      = Am;
00431       Am        = Am_1;
00432       Am_1      = Save;
00433       // EFm:
00434       Save      = EFm;
00435       EFm       = EFm_1;
00436       EFm_1     = Save;
00437       // EBm:
00438       Save      = EBm;
00439       EBm       = EBm_1;
00440       EBm_1     = Save;
00441       Pm_1      = Pm;
00442       
00443       // calc EFm and EBm, Eq.(8):
00444       for (i=1; i<N-m+1; i++)
00445         *(EFm+i) = *(EFm_1+i+1) - *(Am_1+m-1)**(EBm_1+i); 
00446       for (i=0; i<N-m; i++)
00447         *(EBm+i) = *(EBm_1+i)   - std::conj(*(Am_1+m-1))**(EFm_1+i+1);
00448       
00449       // calc Am, Eq.(10):
00450       for (c=CZero,i=0; i<N-1-m; i++)
00451         c+=std::conj(*(EBm+i))**(EFm+i+1);
00452       for (f=0.0,  i=0; i<N-1-m; i++)
00453         f+=mod2(*(EBm+i))+mod2(*(EFm+i+1));
00454       *(Am+m) = 2.0*c/f;
00455       // Eq.(6):
00456       for(i=0; i<m; i++)
00457         *(Am+i) = *(Am_1+i) - *(Am+m)*std::conj(*(Am_1+m-1-i));

00459       // calc Pm, Eq.(7):
