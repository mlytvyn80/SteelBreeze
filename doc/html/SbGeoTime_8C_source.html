00072 int SBMJD::MJD_date(int y, int m, int d) 
00073 {
00074   uint c, ya;
00075   if (y<=99)
00076     y += 1900;
00077   if (m>2)
00078     m -= 3;
00079   else 
00080     {
00081       m += 9;
00082       y--;
00083     };
00084   c = y;
00085   c /= 100;
00086   ya = y - 100*c;
00087   return -678882 + d + (146097*c)/4 + (1461*ya)/4 + (153*m+2)/5;
00088 };

00095 void SBMJD::MJD_rev(int date_, long double time_, 
00096                     int& y, int& m, int& d, int& hr, int& mi, double& se)
00097 {
00098   if (time_>1.0)
00099     {
00100       time_--;
00101       date_++;
00102     };
00103 
00104   uint  x;
00105   uint  j = date_ + 678882;
00106   y = (j*4 - 1)/146097;
00107   j = j*4 - 146097*y - 1;
00108   x = j/4;
00109   j = (x*4 + 3) / 1461;
00110   y = 100*y + j;
00111   x = (x*4) + 3 - 1461*j;
00112   x = (x + 4)/4;
00113   m = (5*x - 3)/153;
00114   x = 5*x - 3 - 153*m;
00115   d = (x + 5)/5;
00116   if (m < 10) 
00117     m += 3; 
00118   else 
00119     {
00120       m -= 9;
00121       y++;
00122     };
00123 
00124   long double tmp = time_*DAY2SEC;
00125   //  se = time_*DAY2SEC;
00126   hr = (int)(tmp/3600.0);
00127   //  hr = (int)(se/3600.0);
00128   mi = (int)((tmp - 3600.0*hr)/60.0);
00129   //  mi = (int)((se - 3600.0*hr)/60.0);
00130   se = tmp - 3600.0*hr - 60.0*mi;
00131   //  se -= 3600.0*hr + 60.0*mi;
00132 };

00634 void SBTime::FundamentalArgs()
00635 {
00636   /*
00637     IERS Technical Note 21,
00638     IERS Conventions (1996)
00639     page 23 says:
00640     ...t is measured in Julian Centuries of 36525 days of 86400 seconds
00641     of Dynamical Time [I hope, "terrestrial"] since J2000.0...
00642    */
00643   double t  = (TT() - TEphem)/36525.0;
00644   //  double t  = (TDB() - TEphem)/36525.0;
00645 
00646   double t2 = t*t;
00647   double t3 = t2*t;
00648   double t4 = t2*t2;
00649  
00650   /* l, the Mean Anomaly of the Moon, arcsecs: */
00651   FArgs[0] = 134.96340251*3600.0 + 1717915923.2178*t + 31.8792*t2 + 0.051635*t3 - 0.00024470*t4;
00652 
00653   /* l', the Mean Anomaly of the Sun, arcsecs: */
00654   // Warning:   sign "-" have been changed to "+" for "0.000136" coeff at t^3
00655   //            according to D.D.McCarthy "Errata for the IERS Conventions (1996)", item 24.
00656   FArgs[1] = 357.52910918*3600.0 + 129596581.0481*t - 0.5532*t2 + 0.000136*t3 - 0.00001149*t4;
00657   // for comp. w/ old ver:
00658   //  FArgs[1] = 1287104.79305 + 129596581.0481*t - 0.5532*t2 + 0.000136*t3 - 0.00001149*t4;
00659   
00660   /* F, Mean Argument of Latitude of the Moon, arcsecs: */
00661   FArgs[2] = 93.27209062*3600.0 + 1739527262.8478*t - 12.7512*t2 - 0.001037*t3 + 0.00000417*t4;
00662 
00663   /* D, the Mean Elongation of the Moon from the Sun, arcsecs: */
00664   FArgs[3] = 297.85019547*3600.0 + 1602961601.2090*t - 6.3706*t2 + 0.006593*t3 - 0.00003169*t4;
00665   //  FArgs[3] = 1072260.70369 + 1602961601.2090*t - 6.3706*t2 + 0.006593*t3 - 0.00003169*t4;
00666 
00667   /* Omega, the Mean Longitude of the Ascending Node of the Moon, arcsecs: */
00668   //                                 - 6962890.5431*t //<- IAU Nut 2000A value
00669   //  FArgs[4] = 125.04455501*3600.0 - 6962890.2665*t + 7.4722*t2 + 0.007702*t3 - 0.00005939*t4;
00670   FArgs[4] = 125.04455501*3600.0 - 6962890.5431*t + 7.4722*t2 + 0.007702*t3 - 0.00005939*t4;
00671   
00672   for (int i=0; i<5; i++) 
00673     {
00674       FArgs[i] = fmod(FArgs[i]*SEC2RAD, 2.0*M_PI); //rads
00675       if (FArgs[i]<0.0) FArgs[i]+=2.0*M_PI; //[0; 2Pi]
00676     };
00677 
00678   /* Earth Rotational Angle = Greenvich Mean Sidereal Time + Pi, rad: */
00680 //  t  = (UT1().date() - TEphem)/36525.0;
00681 //  t2 = t*t;
00682 //  double d = (24110.54841 + 8640184.812866*t + 0.093104*t2 - 6.2e-6*t*t2)/DAY2SEC; //days
00683 //  double r = 1.002737909350795 + 5.9006e-11*t - 5.9e-15*t2;
00684 //  FArgs[5] = 2.0*M_PI*fmod(d + r*UTC().time() + 0.5, 1.0); //rads
00686   double r = 1.00273781191135448;
00687   FArgs[5] = 2.0*M_PI*(0.7790572732640 + fmod(r*(UTC() - TEphem), 1.0)) + M_PI; //rads
00688   if (FArgs[5] < 0.0     )    FArgs[5]+=2.0*M_PI; //[0; 2Pi]
00689   if (FArgs[5] > 2.0*M_PI)    FArgs[5]-=2.0*M_PI; //[0; 2Pi]
00690 
00691   Omega = FArgs[5] - M_PI; // GMST
00692 };
